  <script>
    async function doCheckout() {
      document.getElementById('checkoutBtn').disabled = true;
      document.getElementById('placingOverlay').classList.remove('hidden');

      await new Promise(r => setTimeout(r, 2000));   // 2-second delay

      const res  = await fetch('/checkout', { method: 'POST' });
      const data = await res.json();

      if (data.success) {
        window.location.href = '/order_status';
      } else if (data.waiting) {
        // All 10 slots full – go to waiting room (cart is preserved in session)
        window.location.href = '/waiting_room';
      } else {
        document.getElementById('placingOverlay').classList.add('hidden');
        document.getElementById('checkoutBtn').disabled = false;
        alert(data.message || 'Something went wrong. Please try again.');
      }
    }
  </script>
